// Rename this file to index.ts when deploying with Docker + PostgreSQL.
// Also run: pnpm add drizzle-orm postgres bcryptjs && pnpm add -D drizzle-kit @types/bcryptjs

import { drizzle } from 'drizzle-orm/postgres-js'
import postgres from 'postgres'
import * as schema from './schema'

let _db: ReturnType<typeof drizzle<typeof schema>> | null = null

export function getDb() {
  if (_db) return _db

  const url = process.env.DATABASE_URL
  if (!url) {
    throw new Error(
      'DATABASE_URL is not set. Database features are unavailable in this environment.'
    )
  }

  const conn = postgres(url, { max: 10 })
  _db = drizzle(conn, { schema })
  return _db
}

export { schema }
